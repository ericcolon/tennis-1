model {
  for (i in 1:nm) {
    y[i] ~ dbin(p[i], ns[i])
    logit(p[i]) <- eta[Winner[i], Time[i]] + alpha[Winner[i], Surface[i]] - 
                   eta[Loser[i], Time[i]]  - alpha[Loser[i], Surface[i]]
  }
  
  for (j in 1:np) {
    eta[j,1] ~ dnorm(0, sigma[1]^(-2))
    for (t in 2:T) {
      eta[j,t] ~ dnorm(eta[j,t-1], sigma[2]^(-2))
    }
    for (k in 1:3) {
      alpha[j,k] ~ dnorm(0, sigma[3]^(-2))
    }
  }
  for (j in 1:3) {
    sigma[j] ~ dunif(0, 10)
  }
}
